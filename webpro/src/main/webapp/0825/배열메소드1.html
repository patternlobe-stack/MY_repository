slice(start, end) : start ~ end-1 까지 요소 추출 새로운 배열 생성

splice(index, n,a,b,c....)index위치에서 n개를 삭제하고 a,b,c요소를 삽입
splice(index, 0, a,b,c....) n=0일때는 삭제 하지 않는다 a,b,c요소를 삽입
splice(index, n) : index위치에서 n개를 삭제
arr1 = [1,4];
arr2 = [2,3];

arr1.splice(1, 0, arr2);
console.log(arr1); //arr2가 묶음으로 처리 1 [2,3] 4

arr1.splice(1, 0, ...arr2);
console.log(arr1); //1,2,3,4
